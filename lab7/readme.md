##  Опис схеми

Схеми як передавача так і приймача не змінювалися.

## Код для передавача

``` cpp
#include <VirtualWire.h>
#include "my_keyboard.h"

uint8_t KOUT[] = {9, 8, 7, 6};    // KEYBOARD OUTPUT PINS
uint8_t  KIN[] = {5, 4, 3, 2};
uint8_t msg[] = {'0', '\0'};

void setup() {
  vw_setup(1200);
  vw_set_tx_pin(12);
  keyboard_init(KOUT, KIN);
}

void loop() {
  msg[0] = keyboard_get_char();
  vw_send((uint8_t *)msg, 2);
  vw_wait_tx();
}
```

## Код для приймача
``` cpp
#include "my_SS1D.h"
#include <VirtualWire.h>

uint8_t buf[2] = {0, 0};
uint8_t SEGMENT[] = {26, 24, 22, 23, 25, 27, 28}; // 7S PINS (a, b, c... g, dp)
void setup() {
  vw_setup(1200);
  vw_set_rx_pin(8);
  SS1D_init(SEGMENT);
}
void loop() {
  vw_rx_start();
  if (vw_have_message()) {
    if (vw_get_message(buf, 2)) {
      vw_rx_stop();
    }
  }
  SS1D_write_char(buf[0]);
}
```
